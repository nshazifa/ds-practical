//Picture (angus.JPEG) to HORUS format

from PIL import Image
import pandas as pd
import matplotlib.pyplot as plt
import numpy as np

# Input Agreement
sInputFileName = 'C:/Users/Administrator/Desktop/DS Practical/PRC 2E/Angus.jpg'

# Open the image using Pillow
image = Image.open(sInputFileName).convert('RGBA')
InputData = np.array(image)

# Display the image
plt.imshow(InputData)
plt.show()

# Print the shape of the image data
print('Input Data Values') 
print('X: ', InputData.shape[0])
print('Y: ', InputData.shape[1])
print('RGBA: ', InputData.shape[2])

# Flatten the image data
ProcessRawData = InputData.flatten()

# Calculate new dimensions for reshaping
y = InputData.shape[2] + 2
x = int(ProcessRawData.shape[0] / y)

# Reshape the flattened data and convert to DataFrame
ProcessData = pd.DataFrame(np.reshape(ProcessRawData, (x, y)))

# Assign column names
sColumns = ['XAxis', 'YAxis', 'Red', 'Green', 'Blue', 'Alpha'] 
ProcessData.columns = sColumns
ProcessData

# Print the shape of the processed data
print('Rows: ', ProcessData.shape[0])
print('Columns: ', ProcessData.shape[1])

# Save the processed data to a CSV file
OutputData = ProcessData
OutputData.to_csv('Image_to_HORUS.csv', index=False)
